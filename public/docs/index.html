    Endpoints

    api/user

    Example request:

    curl -X GET \
    -G "http://127.0.0.1:8082/api/user" \
    -H "Content-Type: application/json" \
    -H "Accept: application/json"
    const url = new URL(
    "http://127.0.0.1:8082/api/user"

    Request

    GET api/people

    Store a newly created resource in storage.

    Example request:

    curl -X POST \
    "http://127.0.0.1:8082/api/people" \
    -H "Content-Type: application/json" \
    -H "Accept: application/json" \
    -d '{"preferred_name":"modi","first_name":"provident","middle_name":"atque","last_name":"eligendi","title":"rem","suffix":"veritatis","professional_title":"ipsam","preferred_language":"non","date_of_birth":"hic","person_type_id":8}'

    const url = new URL(
    "http://127.0.0.1:8082/api/people"
);

let headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "preferred_name": "modi",
    "first_name": "provident",
    "middle_name": "atque",
    "last_name": "eligendi",
    "title": "rem",
    "suffix": "veritatis",
    "professional_title": "ipsam",
    "preferred_language": "non",
    "date_of_birth": "hic",
    "person_type_id": 8
}

fetch(url, {
    method: "POST",
    headers: headers,
    body: body
})
    .then(response => response.json())
    .then(json => console.log(json));
